126inline pstsdk::file::file(const std::wstring& filename)
127: m_filename(filename)
128{
129 const char* mode = "rb";
130
131#ifdef _MSC_VER
132 errno_t err = fopen_s(&m_pfile, std::string(filename.begin(), filename.end()).c_str(), mode);
133 if(err != 0)
134 m_pfile = NULL;
135#else
136 m_pfile = fopen(std::string(filename.begin(), filename.end()).c_str(), mode);
137#endif
138 if(m_pfile == NULL)
139 throw std::runtime_error("fopen failed");
140}

148inline size_t pstsdk::file::read(std::vector<byte>& buffer, ulonglong offset) const
149{
150#ifdef _MSC_VER
151 if(_fseeki64(m_pfile, offset, SEEK_SET) != 0)
152#else
153 if(fseek(m_pfile, offset, SEEK_SET) != 0)
154#endif
155 {
156 throw std::out_of_range("fseek failed");
157 }
158
159 size_t read = fread(&buffer[0], 1, buffer.size(), m_pfile);
160
161 if(read != buffer.size())
162 throw std::out_of_range("fread failed");
163
164 return read;
165}

240// We're going to have to do this the hard way, since we don't know how
241// big wchar_t really is, or what encoding it uses.
242#include <iconv.h>
243
244inline std::wstring pstsdk::bytes_to_wstring(const std::vector<byte> &bytes)
245{
246 if(bytes.size() == 0)
247 return std::wstring();
248
249 // Up to one wchar_t for every 2 bytes, if there are no surrogate pairs.
250 if(bytes.size() % 2 != 0)
251 throw std::runtime_error("Cannot interpret odd number of bytes as UTF-16LE");
252 std::wstring out(bytes.size() / 2, L'\0');
253
254 iconv_t cd(::iconv_open("WCHAR_T", "UTF-16LE"));
255 if(cd == (iconv_t)(-1)) {
256 perror("bytes_to_wstring");
257 throw std::runtime_error("Unable to convert from UTF-16LE to wstring");
258 }
259
260 const char *inbuf = reinterpret_cast<const char *>(&bytes[0]);
261 size_t inbytesleft = bytes.size();
262 char *outbuf = reinterpret_cast<char *>(&out[0]);
263 size_t outbytesleft = out.size() * sizeof(wchar_t);
264 size_t result = ::iconv(cd, const_cast<char **>(&inbuf), &inbytesleft, &outbuf, &outbytesleft);
265 ::iconv_close(cd);
266 if(result == (size_t)(-1) || inbytesleft > 0 || outbytesleft % sizeof(wchar_t) != 0)
267 throw std::runtime_error("Failed to convert from UTF-16LE to wstring");
268
269 out.resize(out.size() - (outbytesleft / sizeof(wchar_t)));
270 return out;
271}

273inline std::vector<pstsdk::byte> pstsdk::wstring_to_bytes(const std::wstring &wstr)
274{
275 if(wstr.size() == 0)
276 return std::vector<byte>();
277
278 // Up to 4 bytes per character if all codepoints are surrogate pairs.
279 std::vector<byte> out(wstr.size() * 4);
280
281 iconv_t cd(::iconv_open("UTF-16LE", "WCHAR_T"));
282 if(cd == (iconv_t)(-1)) {
283 perror("wstring_to_bytes");
284 throw std::runtime_error("Unable to convert from wstring to UTF-16LE");
285 }
286
287 const char *inbuf = reinterpret_cast<const char *>(&wstr[0]);
288 size_t inbytesleft = wstr.size() * sizeof(wchar_t);
289 char *outbuf = reinterpret_cast<char *>(&out[0]);
290 size_t outbytesleft = out.size();
291 size_t result = ::iconv(cd, const_cast<char **>(&inbuf), &inbytesleft, &outbuf, &outbytesleft);
292 ::iconv_close(cd);
293 if(result == (size_t)(-1) || inbytesleft > 0)
294 throw std::runtime_error("Failed to convert from wstring to UTF-16LE");
295
296 out.resize(out.size() - outbytesleft);
297 return out;
298}
