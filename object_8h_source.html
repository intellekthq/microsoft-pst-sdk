174template<typename T>
175inline T pstsdk::const_property_object::read_prop(prop_id id) const
176{
177#ifdef _MSC_VER
178#pragma warning(suppress:4127)
179#endif
180 if(!std::is_pod<T>::value)
181 throw std::invalid_argument("T must be a POD or one of the specialized classes");
182
183 if(sizeof(T) == sizeof(ulonglong))
184 {
185 ulonglong t = get_value_8(id);
186 return *(reinterpret_cast<T*>(&t));
187 }
188 else if(sizeof(T) == sizeof(ulong))
189 {
190 ulong t = get_value_4(id);
191 return *(reinterpret_cast<T*>(&t));
192 }
193 else if(sizeof(T) == sizeof(ushort))
194 {
195 ushort t = get_value_2(id);
196 return *(reinterpret_cast<T*>(&t));
197 }
198 else if(sizeof(T) == sizeof(byte))
199 {
200 byte t = get_value_1(id);
201 return *(reinterpret_cast<T*>(&t));
202 }
203 else
204 {
205 std::vector<byte> buffer = get_value_variable(id);
206 return *(reinterpret_cast<T*>(&buffer[0]));
207 }
208}

294template<>
295inline std::vector<std::vector<byte> > const_property_object::read_prop_array<std::vector<byte> >(prop_id id) const
296{
297 std::vector<byte> buffer = get_value_variable(id);
298#ifdef PSTSDK_VALIDATION_LEVEL_WEAK
299 if(buffer.size() < sizeof(ulong))
300 throw std::length_error("mv prop too short");
301#endif
302 disk::mv_toc* ptoc = reinterpret_cast<disk::mv_toc*>(&buffer[0]);
303 std::vector<std::vector<byte> > results;
304
305#ifdef PSTSDK_VALIDATION_LEVEL_WEAK
306 if(buffer.size() < (sizeof(ulong) + ptoc->count * sizeof(ulong)))
307 throw std::length_error("mv prop too short");
308#endif
309
310 for(ulong i = 0; i < ptoc->count; ++i)
311 {
312 ulong start = ptoc->offsets[i];
313 ulong end = (i == (ptoc->count - 1)) ? buffer.size() : ptoc->offsets[i+1];
314#ifdef PSTSDK_VALIDATION_LEVEL_WEAK
315 if(end < start)
316 throw std::length_error("inconsistent mv prop toc");
317#endif
318 results.push_back(std::vector<byte>(&buffer[start], &buffer[start] + (end-start)));
319 }
320
321 return results;
322}
