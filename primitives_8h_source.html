48// Many of the low-level data structures in pstsdk rely on being laid out
49// in the fashion preferred by Visual C++. For example, these structures
50// need to align ulonglong to the nearest 8 bytes. Under GCC, we can force
51// this layout behavior on Mac and Linux systems by appending an
52// appropriate attribute declaration to the structure. We're also tried
53// '#pragma ms_struct on', but it fails on at least some Linux systems.
54#ifdef __GNUC__
55#define PSTSDK_MS_STRUCT __attribute__((ms_struct))
56#else
57#define PSTSDK_MS_STRUCT
58#endif
