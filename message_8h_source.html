411inline pstsdk::message pstsdk::attachment::open_as_message() const
412{
413 if(!is_message())
414 throw std::bad_cast();
415
416 std::vector<byte> buffer = get_bytes();
417 disk::sub_object* psubo = (disk::sub_object*)&buffer[0];
418
419 return message(m_bag.get_node().lookup(psubo->nid));

482{
483 std::wstring buffer = m_bag.read_prop<std::wstring>(PR_SUBJECT_W);
484
485 if(buffer.size() && buffer[0] == message_subject_prefix_lead_byte)
